# 冒泡排序

```js
let arr = [11, 27, 13, 92, 21, 68]
let count = 0;
function msort(_arr) {
    for (let i = 0; i < _arr.length - 1; i++) {
        let done = true; // 如果本次循环没有发生数字交换，说明不需要排序了
        for (let j = 0; j < _arr.length - 1 - i; j++) {
            count++;
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
                done = false
            }
        }
        if (done) {
            break;
        }
    }
}
msort(arr)
console.log(arr);
console.log(count);
```
